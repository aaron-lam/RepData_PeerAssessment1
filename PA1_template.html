total_missing_values <- sum(is.na(data$steps))

# Print the result

Devise a strategy for filling in all of the missing values in the dataset:

data_imputed <- data %>%
  group_by(interval) %>%
  mutate(steps = ifelse(is.na(steps), mean(steps, na.rm = TRUE), steps))
